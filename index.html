class A:
    _attr1 = 1
    _attr2 = "val"

    @property
    def attr1(self):
        return self._attr1
    @attr1.setter
    def attr1(self, val):
        if not isinstance(val, int):
            raise TypeError(...)
        self._attr1 = val
